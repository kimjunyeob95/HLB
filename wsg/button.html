markup 예시

    <!-- 우측위치 -->
    <span class="btn-right"><button type="button" class="btn type01 small">버튼</button><button type="button" class="btn type01 small">버튼</button></span>
    <!-- 좌측위치 -->
    <span class="btn-left"><button type="button" class="btn type01 small">버튼</button><button type="button" class="btn type01 small">버튼</button></span>
</div>

</div>
